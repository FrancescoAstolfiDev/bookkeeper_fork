 91:     @Override
 92:     synchronized CompletableFuture<Boolean> fenceAndLogInJournal(Journal journal) throws IOException {
 93:         boolean success = this.setFenced();
 94:         if (success) {
 95:             // fenced for first time, we should add the key to journal ensure we can rebuild.
 96:             return logFenceEntryInJournal(journal);
 97:         } else {
 98:             // If we reach here, it means the fence state in FileInfo has been set (may not be persisted yet).
 99:             // However, writing the fence log entry to the journal might still be in progress. This can happen
100:             // when a bookie receives two fence requests almost at the same time. The subsequent logic is used
101:             // to check the fencing progress.
102:             if (logFenceResult == null || fenceEntryPersisted.get()){
103:                 // Either ledger's fenced state is recovered from Journal
104:                 // Or Log fence entry in Journal succeed
105:                 CompletableFuture<Boolean> result = FutureUtils.createFuture();
106:                 result.complete(true);
107:                 return result;
108:             } else if (logFenceResult.isDone()) {
109:                 // We failed to log fence entry in Journal, try again.
110:                 return logFenceEntryInJournal(journal);
111:             }
112:             // Fencing is in progress
113:             return logFenceResult;
114:         }
115:     }
116: 
117:     /**
118:      * Log the fence ledger entry in Journal so that we can rebuild the state.
119:      * @param journal log the fence entry in the Journal
120:      * @return A future which will be satisfied when add entry to journal complete
121:      */
122:     private CompletableFuture<Boolean> logFenceEntryInJournal(Journal journal) {
123:         CompletableFuture<Boolean> result;
124:         synchronized (this) {
125:             result = logFenceResult = FutureUtils.createFuture();
126:         }
127:         ByteBuf entry = createLedgerFenceEntry(ledgerId);
128:         try {
129:             journal.logAddEntry(entry, false /* ackBeforeSync */, (rc, ledgerId, entryId, addr, ctx) -> {
130:                 if (LOG.isDebugEnabled()) {
131:                     LOG.debug("Record fenced state for ledger {} in journal with rc {}",
132:                             ledgerId, BKException.codeLogger(rc));
133:                 }
134:                 if (rc == 0) {
135:                     fenceEntryPersisted.compareAndSet(false, true);
136:                     result.complete(true);
137:                 } else {
138:                     result.complete(false);
139:                 }
140:             }, null);
141:         } catch (InterruptedException e) {
142:             Thread.currentThread().interrupt();
143:             result.completeExceptionally(e);
144:         }
145:         return result;
146:     }
